dv.table(
	["","名称","系列","标签"],
	dv.pages('#游戏首页')
	.map(page => {
		const [seriesTag=null] = page.file.etags.filter(tag => tag.includes('系列'))
		const series = seriesTag ? seriesTag : '';
		page.series = series;
		page.gTags = page.file.etags.filter(tag => !tag.includes('系列') && !tag.includes(page.file.name) && !tag.includes('游戏首页') && !page.file.name.includes(tag.replace('#','')) )
		return page;
	})
	.sort(page => [page.series, page.No], 'asc')
	.map(page => [
		page.Icon,
		page.file.link,
		page.series,
		page.gTags.join(' ')
	])
);
~~~

## TGA年度游戏与提名
```ad-quote
title: TGA年度游戏与提名
collapse: close

~~~dataviewjs
dv.table(
	["", "标题","年份","获奖情况"],
	dv.pages("#TGA")
	.map(page => {
		const tgaTags = page.file.etags.filter(tag => tag.includes("TGA"));
		const tga = { year:0, awards:[] }; // year 为获奖年份, awards数组存储所获奖项
		tgaTags.forEach(tag => {
			const [,year,award,result] = tag.split('/')
			tga.year = parseInt(year);
			tga.awards.push({award,result});
		})
		page.tga = tga
		return page;
	})
	.sort(page => [page.tga.year, page.file.name], 'desc')
	.map(page => {
		const award = page.tga.awards.map(item => 
			item.result == '获奖' 
				? item.award == '年度游戏奖' 
					? `==*** ${item.award}: ${item.result} ***==` // 年度游戏 获奖高亮
					: `${item.award}: ${item.result}`
				: `${item.award}: ${item.result}`
		);
		return [
			page.Icon,
			page.file.link,
			page.tga.year,
			award
		]
	})
);
~~~
```
