```ad-quote
title: TGA年度游戏与提名
collapse: close

~~~dataviewjs
dv.table(
	["", "标题","提名年份","奖项"],
	dv.pages("#TGA年度游戏")
	.map(page => {
		const [tagTGA] = page.file.etags.filter(tag => tag.includes("TGA"))
		const [,year,nominate] = tagTGA.split("/")
		page.tga = {
			year,
			got: nominate == undefined ? 0 : 1,
		}
		return page;
	})
	.sort(page => [page.tga.year, page.tga.got], 'asc')
	.map(page => [
		page.Icon,
		page.file.link,
		page.tga.year,
		page.tga.got == 0 ? "年度游戏" : "提名奖"
	])
);
~~~
```
