const order = ["白金","金", "银", "铜"]

		trophy.Type = type;
		return trophy;
	})
	.sort(trophy => [order.indexOf(trophy.Type), trophy.file.name], 'asc')
	.map(trophy => 
		[trophy.Icon, trophy.file.link, trophy.Comment]
	)
);
